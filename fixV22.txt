try {
        // 1. 先更新 food_logs 的內容
        FoodLogDB.update(req.params.id, {
            mealType, calories, protein, carbs, fat, description
        });

        // 2. 取出更新後的日誌，檢查是否有 barcode_id 關聯
        const currentLog = FoodLogDB.findById(req.params.id);

        if (currentLog && currentLog.barcode_id) {
            // 3. 透過 barcode_id 精準更新 barcodes 表
            // 注意：不要更新 barcode 欄位，避免觸發 UNIQUE 衝突
            const stmt = db.prepare(`
                UPDATE barcodes 
                SET name = ?, calories = ?, protein = ?, carbs = ?, fat = ? 
                WHERE id = ?
            `);
            stmt.run(description, calories, protein, carbs, fat, currentLog.barcode_id);
            console.log(`✅ 已精準同步更新 barcodes (ID: ${currentLog.barcode_id})`);
        } else {
            console.log(`⚠️ 此日誌為舊資料無 barcode_id，略過標準食物庫同步。`);
        }

        res.json({ success: true });
        
    } catch (error) {
        console.error('更新食物日誌失敗:', error);
        res.status(500).json({ error: '更新失敗' });
    }
