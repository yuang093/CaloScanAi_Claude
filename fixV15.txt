2. src/middleware/auth.js — 第 3 行：JWT_SECRET 生產環境漏洞

// 生產環境沒設 JWT_SECRET 時只是 warn，之後仍然用預設值 'dev-secret-key'
if (process.env.NODE_ENV === 'production' && !process.env.JWT_SECRET) {
  throw new Error('JWT_SECRET environment variable is required in production');
}


實際上這段 code 會 throw 錯誤，還算有把關。但 Dockerfile 中只有 npm ci，沒有驗證啟動時是否有 JWT_SECRET，若未設定確實會在 runtime 失敗。

Prompt for Fix:
檔案：src/middleware/auth.js
問題：生產環境缺少 JWT_SECRET 會直接 throw error，中斷啟動。但缺點是沒有明確的錯誤訊息告訴使用者如何設定。

建議：保持現有邏輯（throw 是對的），但確認 docker-compose.yml 中 JWT_SECRET 有預設值或預設值夠明顯：
在 docker-compose.yml 的 environment 區塊，JWT_SECRET 預設值改為一個較長的乱码字串，確保生產部署時一定會被置換。

document.addEventListener('DOMContentLoaded', () => {

  } else {
    // 已登入但強迫回到首頁浪費時間，應該驗證 token 有效性後再決定
    // 改為：自動跳過到 dashboard（因為使用者已在登入頁，若已登入代表有 token）
  }
});

4. dashboard.html — 沒有 Auth Guard

// dashboard.html 沒有檢查 token 是否存在、過期
// 任何人都可以直接開啟 dashboard.html


Prompt for Fix:
檔案：dashboard.html
問題：dashboard.html 直接載入，沒有檢查 localStorage 的 token 是否存在與有效。

修改：在 dashboard.html 的 <script> 中，在 DOMContentLoaded 時檢查：
const token = localStorage.getItem('token');
if (!token) {
  window.location.href = 'index.html';
  return;
}

// 額外：驗證 token 格式（是否為有效 JWT）
try {
  const parts = token.split('.');
  if (parts.length !== 3) throw new Error('Invalid JWT format');
} catch (e) {
  localStorage.removeItem('token');
  localStorage.removeItem('user');
  window.location.href = 'index.html';
}

5. src/services/database.js — 密碼驗證時有 Timing Attack 風險

// 第 196-200 行
verifyPassword(username, password) {
  const user = db.prepare('SELECT * FROM users WHERE username = ?').get(username);
  if (!user) return null;  // 用戶不存在就提早返回
  if (!bcrypt.compareSync(password, user.password)) return null;  // 密碼錯誤也提早返回
  return { id: user.id, username: user.username, name: user.name, role: user.role };
}


問題：if (!user) return null; 提早返回可能讓攻擊者根據回應時間推測用戶是否存在（Timing Attack）。建議：不論用戶是否存在，都執行一次 bcrypt.compare，並加上固定的延遲。

Prompt for Fix:
檔案：src/services/database.js
問題：verifyPassword 在用戶不存在時提早返回 bcrypt.compare，可能遭受 Timing Attack。

修改：改為固定時間比對：

verifyPassword(username, password) {
  const user = db.prepare('SELECT * FROM users WHERE username = ?').get(username);
  
  // 固定用一個假的密碼 hash 來比對（不論用戶是否存在都執行）
  // 這樣可以避免 timing attack 推測用戶是否存在
  const fakeHash = '$2a$10$xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx';
  const dummyResult = bcrypt.compareSync(password || '', fakeHash);
  
  if (!user) return null;
  
  if (!bcrypt.compareSync(password, user.password)) return null;
  return { id: user.id, username: user.username, name: user.name, role: user.role };
}
