請幫我重新撰寫 dashboard.html 中的 `downloadShareCard()` 函數。
使用者希望放棄不穩定的 Web Share API 和層級有問題的 Modal，改用「跳出新頁面」的方式讓 iOS 使用者直接長按圖片儲存。

請套用以下邏輯與防阻擋機制：
1. 移除舊有的 `showSaveGuideModal()` 相關邏輯。
2. 在使用者點擊按鈕的「當下（同步）」，立刻呼叫 `window.open('', '_blank')` 開啟一個新分頁，並在裡面寫入「處理中...」的提示，藉此繞過 Safari 的彈出視窗阻擋機制。
3. 等 `html2canvas` 產生完 dataUrl 後，直接覆寫該新分頁的 HTML，設計一個乾淨、帶有按鈕的獨立儲存頁面，確保加入 `-webkit-touch-callout: default;` 讓 iOS 可以順利長按。
4. 加入防呆：如果 `window.open` 依然被阻擋 (例如在 LINE 內建瀏覽器中)，且為 iOS 裝置，則暴力將當前頁面的 `document.body.innerHTML` 替換為圖片，提示使用者儲存後重整網頁。若是 Android/PC，則直接觸發 `<a>` 標籤下載。

請直接套用以下程式碼：

```javascript
async function downloadShareCard() {
  const card = document.getElementById('share-card-preview');
  if (typeof html2canvas === 'undefined') {
    alert('載入中，請稍後再試');
    return;
  }

  const downloadBtn = card.closest('.modal')?.querySelector('[onclick="downloadShareCard()"]');
  if (downloadBtn) {
    downloadBtn.disabled = true;
    downloadBtn.textContent = '處理中...';
  }

  // 【防擋機制】使用者點擊當下「同步」開啟新分頁
  const newWin = window.open('', '_blank');
  if (newWin) {
    newWin.document.write('<meta name="viewport" content="width=device-width, initial-scale=1.0"><h3 style="text-align:center; font-family:sans-serif; margin-top:50px; color:#666;">圖片產生中，請稍候... ⏳</h3>');
  }

  try {
    // 執行耗時的截圖作業
    const canvas = await html2canvas(card, { scale: 2.5, useCORS: true, backgroundColor: null });
    const dataUrl = canvas.toDataURL('image/png');
    const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent) || (navigator.platform === 'MacIntel' && navigator.maxTouchPoints > 1);

    if (newWin) {
      // 成功開啟新分頁，替換成乾淨的儲存畫面
      newWin.document.body.innerHTML = \`
        <div style="display:flex; flex-direction:column; align-items:center; min-height:100vh; background:#fefefe; padding:20px; font-family:sans-serif; box-sizing:border-box;">
          <h2 style="color:#c85a3b; margin-top:10px; margin-bottom:10px;">🎉 產生完成</h2>
          <p style="color:#333; margin-bottom:20px; font-weight:bold; text-align:center; line-height:1.5;">
            👆 這是您的專屬卡片<br>請長按圖片並選擇「儲存影像」
          </p>
          <img src="\${dataUrl}" style="width:100%; max-width:320px; border-radius:12px; box-shadow:0 8px 24px rgba(0,0,0,0.15); -webkit-touch-callout:default; pointer-events:auto;">
          <button onclick="window.close()" style="margin-top:30px; padding:12px 32px; font-size:1rem; border:none; border-radius:24px; background:#e0e0e0; color:#333; cursor:pointer; font-weight:bold;">關閉本頁</button>
        </div>
      \`;
    } else {
      // 如果新分頁不幸被擋 (例如 IG/LINE 內建瀏覽器)
      if (!isIOS) {
        // 非 iOS 直接自動下載
        const fileName = 'caloscanai_' + new Date().toISOString().slice(0,10) + '.png';
        const link = document.createElement('a');
        link.download = fileName;
        link.href = dataUrl;
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
      } else {
        // iOS 最終暴力解：覆蓋原本的畫面，逼迫畫面只剩下圖片
        document.body.innerHTML = \`
          <div style="text-align:center; padding:30px; font-family:sans-serif; background:#fff; min-height:100vh;">
            <h3 style="color:#c85a3b; margin-bottom:16px;">請長按下方圖片儲存</h3>
            <img src="\${dataUrl}" style="width:100%; max-width:350px; border-radius:12px; box-shadow:0 4px 15px rgba(0,0,0,0.2); -webkit-touch-callout:default;">
            <p style="color:#888; margin-top:20px; font-size:0.9rem;">儲存完成後，請重新整理網頁即可恢復原狀</p>
          </div>
        \`;
      }
    }
  } catch (err) {
    console.error('Download error:', err);
    if (newWin) newWin.close();
    alert('圖片產生失敗，請稍後再試');
  } finally {
    if (downloadBtn) {
      downloadBtn.disabled = false;
      downloadBtn.textContent = '📥 下載圖片';
    }
  }
}
